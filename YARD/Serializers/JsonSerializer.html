def serialize(data)

  if YARD::Config.options[:emit_json]
    path = YARD::Config.options[:emit_json]
    log.debug "Serializing json to #{path}"
    File.open!(path, "wb") {|f| f.write data }
  end
  if YARD::Config.options[:emit_json_stdout]
    puts data
  end
end
